\begin{tactic}{congr}
  \begin{tsyntax}[empty]{congr}
  Replace a goal of the form \ec{f t$_1$ ... t$_n$ = f u$_1$ ... u$_n$}
  with the subgoals \ec{t$_i$ = u$_i$} for all \ec{$i$}. Subgoals solvable
  by \ec{reflexivity} are automatically closed.
  \end{tsyntax}
\end{tactic}
