\begin{tactic}{congr}
  \begin{tsyntax}[empty]{congr}
  Replace a goal of the form \tct{f t1 ... tn = f u1 ... un} with the subgoals
  \tct{ti = ui} for all \tct{i}. Subgoals solvable by \tct{reflexivity} are
  automatically closed.
  \end{tsyntax}
\end{tactic}
