\begin{tactic}{congr}
  \begin{tsyntax}[empty]{congr}
  Replace a goal of the form \ec{f t1 ... tn = f u1 ... un} with the subgoals
  \ec{ti = ui} for all \ec{i}. Subgoals solvable by \ec{reflexivity} are
  automatically closed.
  \end{tsyntax}
\end{tactic}
