\begin{tactic}[exact (p : proofterm)]{exact}
  \begin{tsyntax}[empty]{exact}
  Equivalent to \ec{by apply (p : proofterm)}, i.e. apply the given
  proof-term and then try to close the goals with \ec{trivial} - failing
  if not all goals can be closed.
  \end{tsyntax}
\end{tactic}
