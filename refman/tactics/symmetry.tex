\begin{tactic}{symmetry}
  \begin{tsyntax}{symmetry}
  In \prhl, swaps the two programs, transforming the pre and
  postconditions by swapping the memories they refer to.

  \begin{cmathpar}
    \texample[\prhl{}]
      {\ec{symmetry}}
      {\pRHL{P\inv}{c_2}{c_1}{Q\inv}}
      {\pRHL{P}{c_1}{c_2}{Q}}
  \end{cmathpar}
  \end{tsyntax}
\end{tactic}
