\begin{tactic}{skip}
  \begin{tsyntax}{skip} Hoare logic rules for empty statement.

  \textbf{Examples:} 
  \begin{mathpar}
  \inferrule*[left=(\prhl),rightskip=10em]%%
    {P \Rightarrow Q }%%
    {\pRHL{P}{[]}{[]}{Q}}%%
    \quad\raisebox{.7em}{\tct{skip}} \\
  \inferrule*[left=(\phl),rightskip=10em]%%
    {P \Rightarrow Q }%%
    {\pHL{P'}{[]}{Q'}{\diamond}{1}}%%
    \quad\raisebox{.7em}{\tct{skip}} \\
  \inferrule*[left=(\hl),rightskip=10em]%%
    {P \Rightarrow Q}%%
    {\HL{P}{[]}{Q}}%%
    \quad\raisebox{.7em}{\tct{skip}} \\
  \end{mathpar}
  \end{tsyntax}


\end{tactic}
