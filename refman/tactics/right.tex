\begin{tactic}{right}
  \begin{tsyntax}[empty]{right}
    Reduce a goal whose conclusion is a disjunction to one whose
    conclusion is its right member.

    For example, if the current goal is
    \ecinput{examps/parts/tactics/right/1-1.0.ec}{}{}{}{} then
    running \ecinput{examps/parts/tactics/right/1-1.ec}{}{}{}{}
    produces the goal
    \ecinput{examps/parts/tactics/right/1-1.1.ec}{}{}{}{}

    If we replace \ec{\\/} by \ec{||} in this example, we can see the
    difference between the two versions of disjunction: if the current
    goal is \ecinput{examps/parts/tactics/right/2-1.0.ec}{}{}{}{}
    then running
    \ecinput{examps/parts/tactics/right/2-1.ec}{}{}{}{} produces
    the goal \ecinput{examps/parts/tactics/right/2-1.1.ec}{}{}{}{}
  \end{tsyntax}
\end{tactic}
